#ifndef SPLAY_HPP
#define SPLAY_HPP

#include "Node.hpp"

class Splay {
private:
    Node* root;

    Node* zig(Node* x);
    Node* zag(Node* x);
    Node* zigzag(Node* x);
    Node* zagzig(Node* x);
    Node* splay(Node* x);

public:
    Splay();
    ~Splay();

    void insert(char* value);
};

Splay::Splay() : root(nullptr) {}

Splay::~Splay() {
    // liberar memoria
    delete root;
}

// rotación hacia la izquierda
Node* Splay::zig(Node* x) {
    Node* y = x->getNext();
    x->setNext(y->getNext());
    y->setNext(x);
    return y;
}

// rotación hacia la derecha
Node* Splay::zag(Node* x) {
    Node* y = x->getNext();
    x->setNext(y->getNext());
    y->setNext(x);
    return y;
}

// rotación izq-der
Node* Splay::zigzag(Node* x) {
    if ((x == p->getLeft() && p == g->getLeft()) || (x == p->getRight() && p == g->getRight())) {
            if (x == p->getLeft()) {
                // izquierda
                return zig(zig(x));
            } else {
                // derecha
                return zag(zag(x));
            }
        }
}

// rotación der-izq
Node* Splay::zagzig(Node* x) {
    if ((x == p->getLeft() && p == g->getRight()) || (x == p->getRight() && p == g->getLeft())) {
        if (x == p->getLeft()) {
            // izquierda
            return zigzag(x);
        } else {
            // derecha
            return zagzig(x);
        }
    }
}

// splay
Node* Splay::splay(Node* x) {
    if (x == nullptr) {
        return nullptr;
    }

    while (x != root) {
        Node* p = x->getParent();
        if (p == nullptr) {
            // Caso 1: x es hijo del nodo raíz
            if (x == root->getLeft()) {
                root = zig(x);
            } else {
                root = zag(x);
            }
        } else {
            Node* g = p->getParent();
            if (g == nullptr) {
                // Caso 2: x es nieto de la raíz
                if (x == p->getLeft() && p == root->getLeft()) {
                    root = zig(zig(x));
                } else if (x == p->getRight() && p == root->getRight()) {
                    root = zag(zag(x));
                } else if (x == p->getLeft() && p == root->getRight()) {
                    root = zagzig(x);
                } else {
                    root = zigzag(x);
                }
            } else {
                // Caso 3: x es bisnieto de la raíz
                if ((x == p->getLeft() && p == g->getLeft()) || (x == p->getRight() && p == g->getRight())) {
                    if (x == p->getLeft()) {
                        root = zigzig(x);
                    } else {
                        root = zagzag(x);
                    }
                } else {
                    if (x == p->getLeft()) {
                        root = zigzag(x);
                    } else {
                        root = zagzig(x);
                    }
                }
            }
        }
    }

    return root;
}
// inserta el nuevo valor
void Splay::insert(char* value) {
    Node* newNode = new Node(value);
    if (root == nullptr) {
        root = newNode;
    } else {
        list<Node*>* childList = root->getChildsList();
        if (childList->getLength() == 2) {
            Node* leafNode = new Node(value);
            root->getChildsList()->insertar(leafNode);
            root = splay(leafNode);
        } else {
            childList->insertar(newNode);
            root = splay(newNode);
        }
    }
}



#endif
